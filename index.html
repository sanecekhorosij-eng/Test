from PIL import Image from pathlib import Path import zipfile # ==================== НАСТРОЙКИ ==================== src = Path("/mnt/data/6089E3F0-C1F1-4BDE-B566-B25CAF2301EC.jpeg") outdir = Path("/mnt/data/fortune_wheel_prototype") outdir.mkdir(exist_ok=True) # ==================== ОБРАБОТКА ИЗОБРАЖЕНИЯ ==================== img = Image.open(src) # Полный экран img.save(outdir / "screen.jpg", quality=98, optimize=True) # Внутренняя вращающаяся часть колеса (без рамы и указателя) crop_box = (152, 512, 1164, 1524) # square 1012x1012 wheel = img.crop(crop_box) wheel.save(outdir / "wheel.jpg", quality=98, optimize=True) # ==================== HTML ==================== html = r""" Колесо фортуны — прототип
Нажмите «КРУТИТЬ»
Ваша награда
""" (outdir / "index.html").write_text(html, encoding="utf-8") # ==================== README ==================== readme = """КОЛЕСО ФОРТУНЫ — ИНТЕРАКТИВНЫЙ ПРОТОТИП 1. Распакуйте архив. 2. Откройте index.html в браузере. 3. Нажмите кнопку «КРУТИТЬ». Особенности: • Работает полностью оффлайн • Плавная анимация с реалистичным замедлением • Звуковые тики при вращении • Адаптивно под мобильные устройства """ (outdir / "README.txt").write_text(readme, encoding="utf-8") # ==================== СОЗДАНИЕ ZIP ==================== zip_path = Path("/mnt/data/fortune_wheel_prototype.zip") with zipfile.ZipFile(zip_path, "w", zipfile.ZIP_DEFLATED) as z: for p in outdir.iterdir(): z.write(p, arcname=p.name) print(f"✅ Готово!") print(f"Архив создан: {zip_path}") print(f"Файлы: {[p.name for p in outdir.iterdir()]}")
